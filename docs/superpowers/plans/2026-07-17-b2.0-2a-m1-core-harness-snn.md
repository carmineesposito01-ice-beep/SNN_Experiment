```matlab
function info = rtl_gen_dut(blockName, outdir)
%RTL_GEN_DUT  Genera il VHDL di un blocco di snn_champions_lib verso `outdir` (persistente) e
%  ritorna nomi/larghezze delle porte lette dall'ENTITA' generata (non assunti). Il blocco va
%  avvolto in un subsystem (makehdl non gira su un blocco di libreria: my error 2026-07-16).
  here = fileparts(mfilename('fullpath'));
  if nargin < 2, outdir = fullfile(here, ['hdlsrc_' lower(regexprep(blockName,'\W','_'))]); end
  if exist(outdir,'dir'), rmdir(outdir,'s'); end
  bdclose('all');
  lib = 'snn_champions_lib'; load_system(fullfile(here,[lib '.slx']));
  mdl = 'rtlgen_mdl'; if bdIsLoaded(mdl), close_system(mdl,0); end
  new_system(mdl); load_system(mdl);
  add_block([lib '/' blockName], [mdl '/DUT']);
  nIn  = numel(find_system([lib '/' blockName],'SearchDepth',1,'BlockType','Inport'));
  nOut = numel(find_system([lib '/' blockName],'SearchDepth',1,'BlockType','Outport'));
  for j = 1:nIn
    add_block('simulink/Sources/Constant', [mdl '/i' num2str(j)], 'Value','0', ...
              'OutDataTypeStr','fixdt(1,32,20)','SampleTime','1');
    add_line(mdl, ['i' num2str(j) '/1'], ['DUT/' num2str(j)]);
  end
  for j = 1:nOut
    add_block('built-in/Outport', [mdl '/o' num2str(j)], 'Port', num2str(j));
    add_line(mdl, ['DUT/' num2str(j)], ['o' num2str(j) '/1']);
  end
  set_param(mdl,'Solver','FixedStepDiscrete','FixedStep','1','StopTime','10');
  set_param(mdl,'SimulationCommand','update');
  makehdl([mdl '/DUT'], 'TargetLanguage','VHDL', 'TargetDirectory', outdir, ...
          'GenerateHDLTestBench','off');
  vhd = dir(fullfile(outdir,'**','*.vhd'));
  assert(~isempty(vhd), 'nessun VHDL generato per %s', blockName);
  assert(any(strcmp({vhd.name},'DualPortRAM_generic.vhd')), ...
         'manca DualPortRAM -> non e'' l''architettura time-mux');
  ent = fullfile(vhd(strcmp({vhd.name},[blockName '.vhd'])).folder, [blockName '.vhd']);
  txt = fileread(ent);
  info = struct('outdir',outdir,'entity',ent,'nIn',nIn,'nOut',nOut, ...
                'ports', {regexp(txt,'^\s*(\w+)\s*:\s*(IN|OUT)\s+([^;\n]+)','tokens','lineanchors')});
  fprintf('=== VHDL generato: %s (%d file), porte estratte: %d ===\n', ...
          blockName, numel(vhd), numel(info.ports));
  for i=1:numel(info.ports), fprintf('   %-12s %-4s %s\n', info.ports{i}{1},info.ports{i}{2},strtrim(info.ports{i}{3})); end
  close_system(mdl,0); close_system(lib,0);
end
```

```matlab
function rtl_export_vectors(harness, trajList, tag, outdir)
%RTL_EXPORT_VECTORS  Scrive stim/gold .mem + meta per la validazione RTL, dal riferimento MEX veloce.
%  harness 'snn'  -> gold = 5 param IDM (Q7.13, 21b) di Donatello_Champion (rif: snn_decode_lut(...,64)).
%  harness 'ctrl' -> gold = accel (Q4.8, 13b) di Donatello_ACC_IIDM_M (rif: collect_step_mex). [usato in B]
  here = fileparts(mfilename('fullpath'));
  assert(any(strcmp(harness,{'snn','ctrl'})), 'harness: ''snn'' o ''ctrl''');
  ds = load(fullfile(here,'test_dataset.mat')); tr = ds.trajectories;
  d  = load(fullfile(here,'champions_export.mat')); ch = d.champions; if iscell(ch), ch=[ch{:}]; end
  c  = ch(find(arrayfun(@(x) strcmp(char(string(x.name)),'Donatello'),ch),1));
  W  = champ_weights(c); Tp = numerictype(1,21,13);
  if strcmp(harness,'ctrl')
    assert(~isempty(which('collect_step_mex')),'MEX mancante: build_acc_iidm_fsm_mex');
  end
  fs = fopen(fullfile(outdir,['stim_' tag '.mem']),'w');
  fg = fopen(fullfile(outdir,['gold_' tag '.mem']),'w');
  K = 0;
  for t = trajList(:).'
    R   = double(snn_traj_fixed_r16_mex(tr{t}.val, W));
    val = fi(double(tr{t}.val),1,32,20);              % la quantizzazione che vede il blocco
    if strcmp(harness,'ctrl'), clear collect_step_mex; end   % stato OU pulito a inizio traiettoria
    for k = 1:size(val,2)
      for j = 1:4
        fprintf(fs,'%08X\n', typecast(int32(storedInteger(val(j,k))),'uint32'));
      end
      p = snn_decode_lut(fi(R(k,:).',Tp),64);         % 5x1 Q7.13
      if strcmp(harness,'snn')
        for i = 1:5, fprintf(fg,'%06X\n', bitand(uint32(storedInteger(p(i))),uint32(2^21-1))); end
      else
        a = collect_step_mex(double(val(1,k)),double(val(2,k)),double(val(3,k)),double(val(4,k)), ...
                             double(p), k==1);         % accel Q4.8
        af = fi(a,1,13,8);
        fprintf(fg,'%04X\n', bitand(uint32(storedInteger(af)),uint32(2^13-1)));
      end
      K = K + 1;
    end
  end
  fclose(fs); fclose(fg);
  save(fullfile(outdir,['meta_' tag '.mat']),'K','trajList','harness');
  fprintf('rtl_export_vectors(%s): %d control-step su %d traiettorie -> %s\n', harness, K, numel(trajList), outdir);
end
```

- [ ] **Step 1: scrivere il TB `tb_champion_stream.v`**

⚠️ Confermare i nomi porte con l'output della Task 1. Campionamento a **fine control-step** (`HOLD` clock dopo il cambio ingresso: l'uscita è settata e stabile — robusto, non richiede la latenza esatta; `HOLD=500 > ~358`).

```verilog
`timescale 1ns/1ps
// Harness A: pilota Donatello_Champion (VHDL generato) con gli stim, campiona i 5 param a fine
// control-step, confronta col gold. Macro iniettate dal runner: `KVAL, `HOLD, `STIMF, `GOLDF.
module tb_champion_stream;
  localparam integer K    = `KVAL;
  localparam integer HOLD = `HOLD;      // >= latenza RTL (~358); campiona a fine control-step
  reg clk = 0, reset = 0, clk_enable = 1;
  reg  signed [31:0] s, v, dv, v_l;
  wire signed [20:0] v0, T, s0, a, b;   // Q7.13, 21b (confermare nomi/larghezze da Task 1)
  wire ce_out;
  reg  [31:0] stim [0:K*4-1];
  reg  [20:0] gold [0:K*5-1];
  reg  signed [20:0] pw [0:4];
  integer k, i, nmis;

  Donatello_Champion dut (.clk(clk), .reset(reset), .clk_enable(clk_enable),
     .s(s), .v(v), .dv(dv), .v_l(v_l),
     .v0(v0), .T(T), .s0(s0), .a(a), .b(b), .ce_out(ce_out));

  always #62.5 clk = ~clk;              // 8 MHz (mezzo periodo 62.5 ns)

  initial begin
    $readmemh(`STIMF, stim);
    $readmemh(`GOLDF, gold);
    reset = 1; repeat (8) @(posedge clk); reset = 0; repeat (2) @(posedge clk);
    nmis = 0;
    for (k = 0; k < K; k = k + 1) begin
      s = stim[k*4+0]; v = stim[k*4+1]; dv = stim[k*4+2]; v_l = stim[k*4+3];
      repeat (HOLD) @(posedge clk);     // l'edge-trigger avvia l'inferenza; a fine HOLD e' settata
      pw[0]=v0; pw[1]=T; pw[2]=s0; pw[3]=a; pw[4]=b;
      for (i = 0; i < 5; i = i + 1)
        if (pw[i] !== gold[k*5+i]) nmis = nmis + 1;
    end
    $display("RTLRES nMismatch=%0d n=%0d", nmis, K*5);
    $finish;
  end
endmodule
```

```matlab
function res = run_rtl_validate(harness, trajList, mode)
%RUN_RTL_VALIDATE  Orchestratore: genera vettori, invoca xsim, parse RTLRES, assert dei cancelli.
%  mode 'reduced' (default) o 'full'. Ritorna struct con nMismatch/n. Cancello: nMismatch==0.
  if nargin<2 || isempty(trajList), trajList = [1 7 23]; end
  if nargin<3 || isempty(mode), mode = 'reduced'; end
  here = fileparts(mfilename('fullpath'));
  assert(strcmp(harness,'snn'), 'Milestone 1: solo harness ''snn'''); %#ok<*NASGU>
  hdlsrc = fullfile(here,'hdlsrc_champion');
  if ~exist(fullfile(hdlsrc,'Donatello_Champion.vhd'),'file'), rtl_gen_dut('Donatello_Champion', hdlsrc); end
  od = fullfile(here,'axi','champion');
  tag = mode;
  rtl_export_vectors('snn', trajList, tag, od);
  m = load(fullfile(od,['meta_' tag '.mat']));
  HOLD = 500;
  cmd = sprintf('bash run_xsim_champion.sh ../../hdlsrc_champion %d %d stim_%s.mem gold_%s.mem', ...
                m.K, HOLD, tag, tag);
  [rc, out] = system(['cd /d "' od '" && ' cmd]);   %#ok<ASGLU>
  tok = regexp(out, 'RTLRES nMismatch=(\d+) n=(\d+)', 'tokens', 'once');
  assert(~isempty(tok), 'xsim non ha prodotto RTLRES. Output:\n%s', out);
  res = struct('nMismatch', str2double(tok{1}), 'n', str2double(tok{2}), 'K', m.K);
  fprintf('A-1 [%s]: nMismatch = %d / %d (%d control-step x 5 param)\n', mode, res.nMismatch, res.n, res.K);
  assert(res.nMismatch == 0, ['A-1 FALLITO: il VHDL del champion NON riproduce i param di riferimento ' ...
         '(%d/%d disallineati). NON e'' bit-exact a livello RTL.'], res.nMismatch, res.n);
  fprintf('=== A-1 PASSATO: Donatello_Champion RTL bit-exact su %d/%d ===\n', res.n, res.n);
end
```

```matlab
function sensitivity_A1()

         'il cancello e'' CIECO (non confronta davvero).']);

end
```

```matlab
function M = rtl_metrics(trajList)
%RTL_METRICS  Accuratezza di stima dei 5 param IDM (golden vs gt_params) sul dataset. Poiche' A-1
%  prova RTL==golden, queste sono le metriche della versione FPGA. Riporta max e p99 (coda/picco).
  if nargin<1 || isempty(trajList), trajList = [1 7 23]; end
  here = fileparts(mfilename('fullpath'));

  d  = load(fullfile(here,'champions_export.mat')); ch = d.champions; if iscell(ch), ch=[ch{:}]; end
  c  = ch(find(arrayfun(@(x) strcmp(char(string(x.name)),'Donatello'),ch),1));
  W  = champ_weights(c); Tp = numerictype(1,21,13);
  names = {'v0','T','s0','a','b'}; err = [];
  for t = trajList(:).'
    R = double(snn_traj_fixed_r16_mex(tr{t}.val, W)); gt = double(tr{t}.gt_params);
    for k = 1:size(R,1)

    fprintf('  %-3s  max=%.4g  p99=%.4g\n', names{i}, M.(names{i}).max, M.(names{i}).p99);
  end
end
```
