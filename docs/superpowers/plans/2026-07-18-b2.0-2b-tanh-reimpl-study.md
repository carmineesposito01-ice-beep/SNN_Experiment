```matlab
function info = build_tanhdut(tag, srcTanhFile)
%BUILD_TANHDUT  [B2.0-2b L1] Costruisce un DUT scratch con SOLO il tanh (I/O registrato via persistenti,
%  tanh combinatorio in mezzo: replica il path in-blocco st_dd_reg -> tanh -> thl_reg). Chart SOLA nel
%  subsystem (niente blocchi accanto -> niente conversione dataflow, tanh fixed ammesso, HDL_PHASE §9).
%  srcTanhFile = file .m della variante: deve definire `function th = <fname>(dd)` con th sfix19_En17.
  here = fileparts(mfilename('fullpath'));
  out = fullfile(here,'hdl_tanh',tag); if exist(out,'dir'), rmdir(out,'s'); end
  src = fileread(fullfile(here, srcTanhFile));
  fname = regexp(src, 'function\s+\w+\s*=\s*(\w+)\s*\(', 'tokens', 'once'); fname = fname{1};
  main = strjoin({
    'function th = DUT(dd)'          % entry = nome del blocco -> entity DUT (synth top)
    '%#codegen'
    '  persistent ddl thl started'
    '  if isempty(started)'
    ['    ddl = dd; thl = ' fname '(dd); started = true;']
    '  end'
    ['  thl = ' fname '(ddl);   % tanh combinatorio fra i due registri']
    '  ddl = dd;'
    '  th  = thl;'
    'end'}, newline);
  code = [main newline newline src];
  bdclose('all'); mdl = ['m_tanh_' tag]; if bdIsLoaded(mdl), close_system(mdl,0); end
  new_system(mdl); load_system(mdl);
  sub = [mdl '/DUT'];
  add_block('simulink/User-Defined Functions/MATLAB Function', sub);
  ch = sfroot().find('-isa','Stateflow.EMChart','Path',sub); ch.Script = code;
  add_block('simulink/Sources/Constant',[mdl '/i1'],'Value','1','OutDataTypeStr','fixdt(1,19,8)','SampleTime','1');
  add_line(mdl,'i1/1','DUT/1');
  add_block('built-in/Outport',[mdl '/o1'],'Port','1'); add_line(mdl,'DUT/1','o1/1');
  set_param(mdl,'Solver','FixedStepDiscrete','FixedStep','1','StopTime','10');
  set_param(mdl,'SimulationCommand','update');
  makehdl(sub,'TargetLanguage','VHDL','TargetDirectory',out,'GenerateHDLTestBench','off');
  v = dir(fullfile(out,'**','*.vhd')); assert(~isempty(v),'nessun VHDL per %s',tag);
  info = struct('tag',tag,'outdir',v(1).folder,'fname',fname);
  fprintf('L1 DUT %s generato (%d vhd) in %s\n', tag, numel(v), v(1).folder);
end
```

```matlab
function gen_tanh_lut()
%GEN_TANH_LUT  Genera tanh_lut_full.m: tabella 4096 = tanh(fi(x,1,19,8)) su x in [-8,8), sfix19_En17.
%  Bit-exact per costruzione (memoizza lo stesso tanh nativo). Indirizzo = storedInteger(dd)+2048.
  here = fileparts(mfilename('fullpath'));
  s = (-2048:2047).';                         % storedInteger di dd in [-8,8) @ En8
  x = fi(s, 1, 19, 8); x.FractionLength = 8;  % ricostruisci dd
  x = fi(double(s)/256, 1, 19, 8);
  th = tanh(x);                               % nativo, sfix19_En17
  th = fi(th, 1, 19, 17);
  vals = storedInteger(th);                   % 4096 interi (sfix19_En17)
  fid = fopen(fullfile(here,'tanh_lut_full.m'),'w');
  fprintf(fid, 'function th = tanh_lut_full(dd) %%#codegen\n');
  fprintf(fid, '%% [B2.0-2b A1] tanh via LUT PIENA bit-exact: memoizza tanh(fi(x,1,19,8)) su [-8,8).\n');
  fprintf(fid, '%%  Generato da gen_tanh_lut.m -- NON modificare a mano. th: sfix19_En17.\n');
  fprintf(fid, '  persistent TBL\n  if isempty(TBL)\n    TBL = int32([ ...\n');
  fprintf(fid, '      %d %d %d %d %d %d %d %d ...\n', vals);   % 4096 literal
  fprintf(fid, '    ]);\n  end\n');
  fprintf(fid, '  s = int32(storedInteger(dd));           %% dd*2^8, intero\n');
  fprintf(fid, '  if s < -2048\n    a = int32(1);\n  elseif s > 2047\n    a = int32(4096);\n  else\n    a = s + int32(2049);\n  end\n');
  fprintf(fid, '  th = reinterpretcast(fi(TBL(a),1,19,0), numerictype(1,19,17));\n');
  fprintf(fid, 'end\n');
  fclose(fid);
  fprintf('scritto tanh_lut_full.m (4096 entry)\n');
end
```
(Nota impl.: `reinterpretcast` porta l'intero memorizzato al tipo sfix19_En17 senza cambiare i bit. Se `reinterpretcast` desse problemi in HDL, alternativa: `th = fi(double(TBL(a))/2^17,1,19,17)` — ma verificare bit-identità nel dmax.)

```matlab
function th = tanh_lut_interp(dd) %#codegen
% [B2.0-2b A2a] tanh via LUT PICCOLA (256) + interpolazione lineare su [-8,8). Approssimato (NON bit-exact).
  persistent TBL
  if isempty(TBL)
    xs = fi(linspace(-8,8,256).',1,19,8);
    TBL = fi(tanh(xs),1,19,17);           % 256 nodi
  end
  ddc = dd; if ddc<-8, ddc(:)=-8; end; if ddc>=8, ddc(:)=fi(8-1/256,1,19,8); end
  u = (ddc + 8) * (255/16);               % posizione continua in [0,255]
  i0 = int32(floor(u)); if i0>254, i0=int32(254); end
  fr = u - double(i0);
  th = fi(TBL(i0+1) + (TBL(i0+2)-TBL(i0+1))*fr, 1, 19, 17);
end
```

## Task 5: A2c — CORDIC iperbolico

**Files:**
- Create: `matlab/tanh_cordic.m`

- [ ] **Step 1: `matlab/tanh_cordic.m`** (CORDIC iperbolico, `tanh=sinh/cosh`, con range-reduction)

```matlab
function th = tanh_cordic(dd) %#codegen
% [B2.0-2b A2c] tanh via CORDIC iperbolico. Range-reduction: dd = k*ln2 + r, tanh via sinh/cosh(r) e
%  formule di somma; qui versione diretta con clamp (il CORDIC iperbolico converge ~|r|<1.13). Approssimato.
  ddc = dd; if ddc<-8, ddc(:)=-8; end; if ddc>=8, ddc(:)=fi(8-1/256,1,19,8); end
  % riduzione: usa la simmetria e satura oltre |x|>3 (tanh~±1 con err<1e-2, sufficiente per la CARATTERIZZAZIONE)
  x = double(ddc);
  if x >  3, th = fi( tanh(3),1,19,17); return; end
  if x < -3, th = fi(-tanh(3),1,19,17); return; end
  % CORDIC iperbolico (14 iterazioni, ripetendo i=4,13): calcola sinh,cosh di x
  ang = [0.549306 0.255413 0.125657 0.062582 0.031260 0.015626 0.007813 0.003906 0.001953 0.000977 0.000488 0.000244 0.000122 0.000061];
  cN  = 1.207497;                          % gain iperbolico inverso (prodotto 1/sqrt(1-2^-2i))
  sh=0.0; chh=cN; z=x; rep=4;
  for i=1:14
    ii=i; if ii>rep, ii=ii; end
    di = (z>=0)*2-1;                        % sign
    f = 2^-(min(i,13));
    sh2  = sh  + di*chh*f;
    chh2 = chh + di*sh *f;
    z    = z   - di*ang(i);
    sh=sh2; chh=chh2;
  end
  th = fi(sh/chh, 1, 19, 17);
end
```
